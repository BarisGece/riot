1   package com.redislabs.riot;
2   
3   import com.redislabs.mesclun.search.RediSearchUtils;
4   import com.redislabs.riot.convert.RegexNamedGroupsExtractor;
5   import com.redislabs.riot.processor.*;
6   import lombok.Data;
7   import org.springframework.batch.item.ItemProcessor;
8   import org.springframework.core.convert.converter.Converter;
9   import org.springframework.expression.EvaluationContext;
10  import org.springframework.expression.Expression;
11  import org.springframework.expression.spel.support.StandardEvaluationContext;
12  import org.springframework.util.ObjectUtils;
13  import picocli.CommandLine.Option;
14  
15  import java.text.SimpleDateFormat;
16  import java.util.*;
17  
18  @Data
19  public class ProcessorOptions {
20  
21      @Option(arity = "1..*", names = "--process", description = "SpEL expressions in the form field1=\"exp\" field2=\"exp\"...", paramLabel = "<f=exp>")
22      private Map<String, Expression> spelFields;
23      @Option(arity = "1..*", names = "--var", description = "Register a variable in the SpEL processor context.", paramLabel = "<v=exp>")
24      private Map<String, Expression> variables;
25      @Option(names = "--date", description = "Processor date format (default: ${DEFAULT-VALUE}).", paramLabel = "<string>")
26      private String dateFormat = new SimpleDateFormat().toPattern();
27      @Option(arity = "1..*", names = "--filter", description = "Discard records using SpEL boolean expressions.", paramLabel = "<exp>")
28      private String[] filters;
29      @Option(arity = "1..*", names = "--regex", description = "Extract named values from source field using regex.", paramLabel = "<f=exp>")
30      private Map<String, String> regexes;
31  
32      public ItemProcessor<Map<String, Object>, Map<String, Object>> processor(RedisOptions redisOptions) throws NoSuchMethodException {
33          List<ItemProcessor<Map<String, Object>, Map<String, Object>>> processors = new ArrayList<>();
34          if (!ObjectUtils.isEmpty(spelFields)) {
35              StandardEvaluationContext context = new StandardEvaluationContext();
36              context.setVariable("date", dateFormat);
37              processors.add(new SpelProcessor(redisOptions, context(), spelFields));
38          }
39          if (!ObjectUtils.isEmpty(regexes)) {
40              Map<String, Converter<String, Map<String, String>>> fields = new LinkedHashMap<>();
41              regexes.forEach((f, r) -> fields.put(f, RegexNamedGroupsExtractor.builder().regex(r).build()));
42              processors.add(new MapProcessor(fields));
43          }
44          if (!ObjectUtils.isEmpty(filters)) {
45              processors.add(new FilteringProcessor(filters));
46          }
47          if (processors.isEmpty()) {
48              return null;
49          }
50          if (processors.size() == 1) {
51              return processors.get(0);
52          }
53          CompositeItemStreamItemProcessor<Map<String, Object>, Map<String, Object>> compositeItemProcessor = new CompositeItemStreamItemProcessor<>();
54          compositeItemProcessor.setDelegates(processors);
55          return compositeItemProcessor;
56      }
57  
58      private EvaluationContext context() throws NoSuchMethodException {
59          StandardEvaluationContext context = new StandardEvaluationContext();
60          context.setVariable("date", new SimpleDateFormat(dateFormat));
61          if (variables != null) {
62              for (String variable : variables.keySet()) {
63                  context.setVariable(variable, variables.get(variable).getValue(context));
64              }
65          }
66          context.registerFunction("geo", RediSearchUtils.GeoLocation.class.getDeclaredMethod("toString", String.class, String.class));
67          context.setPropertyAccessors(Collections.singletonList(new MapAccessor()));
68          return context;
69      }

72  }
