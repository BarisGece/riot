36  @EqualsAndHashCode(callSuper = true)
37  @Command(name = "import", description = "Import Kafka topics into Redis streams")
38  public class StreamImportCommand extends AbstractFlushingTransferCommand {
39  
40      @SuppressWarnings("unused")
41      @Parameters(arity = "0..*", description = "One ore more topics to read from", paramLabel = "TOPIC")
42      private String[] topics;
43      @CommandLine.Mixin
44      private KafkaOptionsons.html#KafkaOptions">KafkaOptions options = new KafkaOptions();
45      @SuppressWarnings("unused")
46      @Option(names = "--key", description = "Target stream key (default: same as topic)", paramLabel = "<string>")
47      private String key;
48      @SuppressWarnings("unused")
49      @Option(names = "--maxlen", description = "Stream maxlen", paramLabel = "<int>")
50      private Long maxlen;
51      @SuppressWarnings("unused")
52      @Option(names = "--trim", description = "Stream efficient trimming ('~' flag)")
53      private boolean approximateTrimming;
54      @CommandLine.Mixin
55      private FilteringOptionsilteringOptions">FilteringOptions filteringOptions = new FilteringOptions();
56  
57      @Override
58      protected Flow flow(StepBuilderFactory stepBuilderFactory) {
59          Assert.isTrue(!ObjectUtils.isEmpty(topics), "No topic specified");
60          List<Step> steps = new ArrayList<>();
61          Properties consumerProperties = options.consumerProperties();
62          log.debug("Using Kafka consumer properties: {}", consumerProperties);
63          for (String topic : topics) {
64              log.debug("Creating Kafka reader for topic {}", topic);
65              KafkaItemReader<String, Object> reader = new KafkaItemReaderBuilder<String, Object>().partitions(0).consumerProperties(consumerProperties).partitions(0).name(topic).saveState(false).topic(topic).build();
66              StepBuilder stepBuilder = stepBuilderFactory.get(topic + "-stream-import-step");
67              RiotStepBuilder<ConsumerRecord<String, Object>, ConsumerRecord<String, Object>> step = riotStep(stepBuilder, "Importing from " + topic);
68              steps.add(configure(step.reader(reader).writer(writer()).build()).build());
69          }
70          return flow(steps.toArray(new Step[0]));
71      }
72  
73      private ItemWriter<ConsumerRecord<String, Object>> writer() {
74          OperationItemWriter.RedisOperation<ConsumerRecord<String, Object>> operation = new Xadd<>(keyConverter(), bodyConverter(), xAddArgs());
75          RedisOptions redisOptions = getRedisOptions();
76          if (redisOptions.isCluster()) {
77              log.debug("Creating cluster stream writer");
78              return OperationItemWriter.operation(operation).client(redisOptions.redisClusterClient()).poolConfig(redisOptions.poolConfig()).build();
79          }
80          log.debug("Creating stream writer");
81          return OperationItemWriter.operation(operation).client(redisOptions.redisClient()).poolConfig(redisOptions.poolConfig()).build();
82      }
83  
84      private Converter<ConsumerRecord<String, Object>, Map<String, String>> bodyConverter() {
85          if (options.getSerde() == KafkaOptions.SerDe.JSON) {
86              return new JsonToMapConverter(filteringOptions.converter());
87          }
88          return new AvroToMapConverter(filteringOptions.converter());
89      }
90  
91      private Converter<ConsumerRecord<String, Object>, String> keyConverter() {
92          if (key == null) {
93              return ConsumerRecord::topic;
94          }
95          return s -> key;
96      }
97  
98      private XAddArgs xAddArgs() {
99          if (maxlen == null) {
100             return null;
101         }
102         XAddArgs args = new XAddArgs();
103         args.maxlen(maxlen);
104         args.approximateTrimming(approximateTrimming);
105         return args;
106     }
