1   package com.redislabs.riot.stream.processor;
2   
3   import org.apache.avro.Schema;
4   import org.apache.avro.SchemaBuilder;
5   import org.apache.avro.SchemaBuilder.FieldAssembler;
6   import org.apache.avro.SchemaBuilder.RecordBuilder;
7   import org.apache.avro.generic.GenericData;
8   import org.apache.avro.generic.GenericRecord;
9   import org.springframework.core.convert.converter.Converter;
10  
11  import io.lettuce.core.StreamMessage;
12  
13  public class AvroProducerProcessor extends AbstractProducerProcessor {
14  
15  	private Schema schema;
16  
17  	public AvroProducerProcessor(Converter<StreamMessage<String, String>, String> topicConverter) {
18  		super(topicConverter);
19  	}
20  
21  	@Override
22  	protected GenericRecord value(StreamMessage<String, String> message) {
23  		if (schema == null) {
24  			RecordBuilder<Schema> builder = SchemaBuilder.record(message.getStream());
25  			FieldAssembler<Schema> fields = builder.fields();
26  			for (String fieldName : message.getBody().keySet()) {
27  				fields.name(fieldName).type().optional().stringType();
28  			}
29  			this.schema = fields.endRecord();
30  		}
31  		GenericRecord record = new GenericData.Record(schema);
32  		message.getBody().forEach(record::put);
33  		return record;
34  	}
35  
36  }
