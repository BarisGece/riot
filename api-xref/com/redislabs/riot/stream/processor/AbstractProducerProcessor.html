1   package com.redislabs.riot.stream.processor;
2   
3   import io.lettuce.core.StreamMessage;
4   import org.apache.kafka.clients.producer.ProducerRecord;
5   import org.springframework.batch.item.ItemProcessor;
6   import org.springframework.core.convert.converter.Converter;
7   
8   public abstract class AbstractProducerProcessor
9           implements ItemProcessor<StreamMessage<String, String>, ProducerRecord<String, Object>> {
10  
11      private final Converter<StreamMessage<String, String>, String> topicConverter;
12  
13      protected AbstractProducerProcessor(Converter<StreamMessage<String, String>, String> topicConverter) {
14          this.topicConverter = topicConverter;
15      }
16  
17      @Override
18      public ProducerRecord<String, Object> process(StreamMessage<String, String> item) {
19          String topic = topicConverter.convert(item);
20          if (topic == null) {
21              throw new IllegalStateException("Topic is null");
22          }
23          return new ProducerRecord<>(topic, value(item));
24      }
25  
26      protected abstract Object value(StreamMessage<String, String> body);
27  
28  }
