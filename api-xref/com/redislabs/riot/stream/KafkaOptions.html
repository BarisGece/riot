1   package com.redislabs.riot.stream;
2   
3   import io.confluent.kafka.serializers.AbstractKafkaSchemaSerDeConfig;
4   import io.confluent.kafka.serializers.KafkaAvroDeserializer;
5   import io.confluent.kafka.serializers.KafkaAvroSerializer;
6   import lombok.Data;
7   import org.apache.kafka.common.serialization.Deserializer;
8   import org.apache.kafka.common.serialization.Serializer;
9   import org.apache.kafka.common.serialization.StringDeserializer;
10  import org.apache.kafka.common.serialization.StringSerializer;
11  import org.springframework.boot.autoconfigure.kafka.KafkaProperties;
12  import org.springframework.boot.autoconfigure.kafka.KafkaProperties.Consumer;
13  import org.springframework.boot.autoconfigure.kafka.KafkaProperties.Producer;
14  import org.springframework.kafka.support.serializer.JsonDeserializer;
15  import org.springframework.kafka.support.serializer.JsonSerializer;
16  import org.springframework.util.ObjectUtils;
17  import org.springframework.util.unit.DataSize;
18  import picocli.CommandLine.Option;
19  
20  import java.util.Arrays;
21  import java.util.LinkedHashMap;
22  import java.util.Map;
23  import java.util.Properties;
24  
25  @Data
26  public class KafkaOptions {
27  
28      public enum SerDe {
29          AVRO, JSON
30      }
31  
32      @Option(arity = "1..*", names = "--broker", description = "One or more brokers in the form host:port.", paramLabel = "<server>")
33      private String[] brokers;
34      @Option(names = "--group", description = "Consumer group id.", paramLabel = "<id>")
35      private String groupId = "$Default";
36      @Option(names = "--registry", description = "Schema registry URL.", paramLabel = "<url>")
37      private String schemaRegistryUrl;
38      @Option(arity = "1..*", names = {"-p", "--property"}, description = "Additional producer/consumer properties.", paramLabel = "<k=v>")
39      private Map<String, String> properties;
40      @Option(names = "--serde", description = "Serializer/Deserializer: ${COMPLETION-CANDIDATES} (default: ${DEFAULT-VALUE}).", paramLabel = "<serde>")
41      private SerDe serde = SerDe.JSON;
42  
43      public Properties consumerProperties() {
44          KafkaProperties kafkaProperties = kafkaProperties();
45          Consumer consumer = kafkaProperties.getConsumer();
46          consumer.setGroupId(groupId);
47          consumer.setKeyDeserializer(StringDeserializer.class);
48          consumer.setValueDeserializer(deserializer());
49          Properties consumerProperties = new Properties();
50          consumerProperties.putAll(kafkaProperties.buildConsumerProperties());
51          consumerProperties.putAll(properties());
52          if (serde == SerDe.JSON) {
53              consumerProperties.put(JsonDeserializer.VALUE_DEFAULT_TYPE, Map.class);
54          }
55          return consumerProperties;
56      }
57  
58      private Map<String, Object> properties() {
59          Map<String, Object> properties = new LinkedHashMap<>();
60          if (schemaRegistryUrl != null) {
61              properties.put(AbstractKafkaSchemaSerDeConfig.SCHEMA_REGISTRY_URL_CONFIG, schemaRegistryUrl);
62          }
63          if (!ObjectUtils.isEmpty(this.properties)) {
64              properties.putAll(this.properties);
65          }
66          return properties;
67      }
68  
69      private KafkaProperties kafkaProperties() {
70          KafkaProperties kafkaProperties = new KafkaProperties();
71          if (!ObjectUtils.isEmpty(brokers)) {
72              kafkaProperties.setBootstrapServers(Arrays.asList(brokers));
73          }
74          return kafkaProperties;
75      }
76  
77      public Map<String, Object> producerProperties() {
78          KafkaProperties kafkaProperties = kafkaProperties();
79          Producer producer = kafkaProperties.getProducer();
80          producer.setBatchSize(DataSize.ofBytes(16384));
81          producer.setBufferMemory(DataSize.ofBytes(33554432));
82          producer.setKeySerializer(StringSerializer.class);
83          producer.setValueSerializer(serializer());
84          Map<String, Object> producerProperties = properties();
85          producerProperties.putAll(kafkaProperties.buildProducerProperties());
86          return producerProperties;
87      }
88  
89      @SuppressWarnings("rawtypes")
90      public Class<? extends Serializer> serializer() {
91          switch (serde) {
92              case JSON:
93                  return JsonSerializer.class;
94              default:
95                  return KafkaAvroSerializer.class;
96          }
97      }
98  
99      @SuppressWarnings("rawtypes")
100     private Class<? extends Deserializer> deserializer() {
101         switch (serde) {
102             case JSON:
103                 return JsonDeserializer.class;
104             default:
105                 return KafkaAvroDeserializer.class;
106         }
107     }
108 
109 }
