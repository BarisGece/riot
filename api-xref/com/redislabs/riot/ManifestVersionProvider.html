1   package com.redislabs.riot;
2   
3   import picocli.CommandLine;
4   
5   import java.io.IOException;
6   import java.net.URL;
7   import java.util.Enumeration;
8   import java.util.jar.Attributes;
9   import java.util.jar.Manifest;
10  
11  /**
12   * {@link picocli.CommandLine.IVersionProvider} implementation that returns version information from
13   * the jar file's {@code /META-INF/MANIFEST.MF} file.
14   */
15  public class ManifestVersionProvider implements CommandLine.IVersionProvider {
16  
17      @Override
18      public String[] getVersion() {
19          return new String[]{
20                  // @formatter:off
21                  "",
22                  "      ▀        █     @|fg(4;1;1) ██████████████████████████|@",
23                  " █ ██ █  ███  ████   @|fg(4;2;1) ██████████████████████████|@",
24                  " ██   █ █   █  █     @|fg(5;4;1) ██████████████████████████|@",
25                  " █    █ █   █  █     @|fg(1;4;1) ██████████████████████████|@",
26                  " █    █  ███    ██   @|fg(0;3;4) ██████████████████████████|@"+ "  v" + getVersionString(),
27                  ""};
28                  // @formatter:on
29      }
30  
31      private String getVersionString() {
32          try {
33              Enumeration<URL> resources = getClass().getClassLoader().getResources("META-INF/MANIFEST.MF");
34              while (resources.hasMoreElements()) {
35                  URL url = resources.nextElement();
36                  Manifest manifest = new Manifest(url.openStream());
37                  if (isApplicableManifest(manifest)) {
38                      Attributes attr = manifest.getMainAttributes();
39                      return String.valueOf(get(attr, "Implementation-Version"));
40                  }
41              }
42          } catch (IOException ex) {
43              // ignore
44          }
45          return "N/A";
46      }
47  
48      private boolean isApplicableManifest(Manifest manifest) {
49          Attributes attributes = manifest.getMainAttributes();
50          return "RIOT".equals(get(attributes, "Implementation-Title"));
51      }
52  
53      private static Object get(Attributes attributes, String key) {
54          return attributes.get(new Attributes.Name(key));
55      }
56  }
