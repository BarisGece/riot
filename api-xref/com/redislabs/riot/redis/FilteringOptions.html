1   package com.redislabs.riot.redis;
2   
3   import com.redislabs.riot.convert.CompositeConverter;
4   import com.redislabs.riot.convert.MapFilteringConverter;
5   import com.redislabs.riot.convert.MapFlattener;
6   import com.redislabs.riot.convert.ObjectToStringConverter;
7   import lombok.Data;
8   import org.springframework.core.convert.converter.Converter;
9   import org.springframework.util.ObjectUtils;
10  import picocli.CommandLine;
11  
12  import java.util.Map;
13  
14  @Data
15  public class FilteringOptions {
16  
17      @CommandLine.Option(arity = "1..*", names = "--include", description = "Fields to include", paramLabel = "<field>")
18      private String[] includes;
19      @CommandLine.Option(arity = "1..*", names = "--exclude", description = "Fields to exclude", paramLabel = "<field>")
20      private String[] excludes;
21  
22      public Converter<Map<String, Object>, Map<String, String>> converter() {
23          MapFlattener<String> mapFlattener = new MapFlattener<>(new ObjectToStringConverter());
24          if (ObjectUtils.isEmpty(includes) && ObjectUtils.isEmpty(excludes)) {
25              return mapFlattener;
26          }
27          MapFilteringConverter.MapFilteringConverterBuilder<String, Object> filtering = MapFilteringConverter.builder();
28          if (!ObjectUtils.isEmpty(includes)) {
29              filtering.includes(includes);
30          }
31          if (!ObjectUtils.isEmpty(excludes)) {
32              filtering.excludes(excludes);
33          }
34          return new CompositeConverter(mapFlattener, filtering.build());
35      }
36  
37  
38  }
