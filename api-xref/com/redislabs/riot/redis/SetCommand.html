1   package com.redislabs.riot.redis;
2   
3   import com.fasterxml.jackson.annotation.JsonInclude.Include;
4   import com.fasterxml.jackson.databind.ObjectMapper;
5   import com.fasterxml.jackson.databind.SerializationFeature;
6   import com.fasterxml.jackson.dataformat.xml.XmlMapper;
7   import com.redislabs.riot.convert.ObjectMapperConverter;
8   import lombok.extern.slf4j.Slf4j;
9   import org.springframework.batch.item.redis.OperationItemWriter;
10  import org.springframework.batch.item.redis.support.operation.Set;
11  import org.springframework.core.convert.converter.Converter;
12  import picocli.CommandLine.Command;
13  import picocli.CommandLine.Option;
14  
15  import java.util.Map;
16  
17  @Slf4j
18  @Command(name = "set", description = "Set strings from input")
19  public class SetCommand extends AbstractKeyCommand {
20  
21      private enum StringFormat {
22          RAW, XML, JSON
23      }
24  
25      @Option(names = "--format", description = "Serialization: ${COMPLETION-CANDIDATES} (default: ${DEFAULT-VALUE})", paramLabel = "<fmt>")
26      private StringFormat format = StringFormat.JSON;
27      @SuppressWarnings("unused")
28      @Option(names = "--field", description = "Raw value field", paramLabel = "<field>")
29      private String field;
30      @SuppressWarnings("unused")
31      @Option(names = "--root", description = "XML root element name", paramLabel = "<name>")
32      private String root;
33  
34      @Override
35      public OperationItemWriter.RedisOperation<Map<String, Object>> operation() {
36          return new Set<>(key(), stringValueConverter());
37      }
38  
39      private Converter<Map<String, Object>, String> stringValueConverter() {
40          switch (format) {
41              case RAW:
42                  if (field == null) {
43                      log.warn("Raw value field name not set");
44                  }
45                  return stringFieldExtractor(field);
46              case XML:
47                  return new ObjectMapperConverter<>(new XmlMapper().writer().withRootName(root));
48              default:
49                  ObjectMapper jsonMapper = new ObjectMapper();
50                  jsonMapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
51                  jsonMapper.setSerializationInclusion(Include.NON_NULL);
52                  return new ObjectMapperConverter<>(jsonMapper.writer().withRootName(root));
53          }
54      }
55  
56  }
