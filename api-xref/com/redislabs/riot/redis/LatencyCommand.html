1   package com.redislabs.riot.redis;
2   
3   import io.lettuce.core.api.sync.BaseRedisCommands;
4   import io.lettuce.core.metrics.CommandMetrics;
5   import io.lettuce.core.metrics.DefaultCommandLatencyCollectorOptions;
6   import org.HdrHistogram.Histogram;
7   import org.LatencyUtils.LatencyStats;
8   import picocli.CommandLine.Command;
9   import picocli.CommandLine.Option;
10  
11  import java.util.Map;
12  import java.util.TreeMap;
13  import java.util.concurrent.TimeUnit;
14  
15  @Command(name = "latency", description = "Calculate latency stats")
16  public class LatencyCommand extends AbstractUtilityCommand {
17  
18  	@Option(names = "--iterations", description = "Number of latency tests (default: ${DEFAULT-VALUE})", paramLabel = "<count>")
19  	private int iterations = 1000;
20  	@Option(names = "--sleep", description = "Sleep duration between calls (default: ${DEFAULT-VALUE})", paramLabel = "<ms>")
21  	private long sleep = 1;
22  	@Option(names = "--unit", description = "Latency unit (default: ${DEFAULT-VALUE})", paramLabel = "<unit>")
23  	private TimeUnit unit = TimeUnit.MILLISECONDS;
24  	@Option(names = "--show-distribution", description = "Show latency distribution")
25  	private boolean showDistribution = false;
26  
27  	@Override
28  	protected void execute(BaseRedisCommands<String, String> commands) {
29  		LatencyStats stats = new LatencyStats();
30  		for (int index = 0; index < iterations; index++) {
31  			long startTime = System.nanoTime();
32  			commands.ping();
33  			stats.recordLatency(System.nanoTime() - startTime);
34  			try {
35  				Thread.sleep(sleep);
36  			} catch (InterruptedException e) {
37  				return;
38  			}
39  		}
40  		Histogram histogram = stats.getIntervalHistogram();
41  		if (showDistribution) {
42  			histogram.outputPercentileDistribution(System.out, 1000000.0);
43  		} else {
44  			DefaultCommandLatencyCollectorOptions options = DefaultCommandLatencyCollectorOptions.create();
45  			Map<Double, Long> percentiles = new TreeMap<>();
46  			for (double targetPercentile : options.targetPercentiles()) {
47  				percentiles.put(targetPercentile,
48  						unit.convert(histogram.getValueAtPercentile(targetPercentile), TimeUnit.NANOSECONDS));
49  			}
50  			CommandMetrics.CommandLatency latency = new CommandMetrics.CommandLatency(
51  					unit.convert(histogram.getMinValue(), TimeUnit.NANOSECONDS),
52  					unit.convert(histogram.getMaxValue(), TimeUnit.NANOSECONDS), percentiles);
53  			System.out.println(latency);
54  		}
55  	}
56  }
