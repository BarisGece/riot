1   package com.redislabs.riot.redis;
2   
3   import com.redislabs.mesclun.api.async.RedisModulesAsyncCommands;
4   import com.redislabs.mesclun.search.SugaddOptions;
5   import io.lettuce.core.RedisFuture;
6   import io.lettuce.core.api.async.BaseRedisAsyncCommands;
7   import org.springframework.batch.item.redis.support.operation.AbstractKeyOperation;
8   import org.springframework.core.convert.converter.Converter;
9   import org.springframework.util.Assert;
10  
11  public class Sugadd<T> extends AbstractKeyOperation<T, String> {
12  
13      private final Converter<T, Double> score;
14      private final Converter<T, String> payload;
15      private final boolean increment;
16  
17      public Sugadd(Converter<T, String> key, Converter<T, String> string, Converter<T, Double> score, Converter<T, String> payload, boolean increment) {
18          super(key, string, t -> false);
19          Assert.notNull(string, "A string converter is required");
20          Assert.notNull(score, "A score converter is required");
21          this.score = score;
22          this.payload = payload;
23          this.increment = increment;
24      }
25  
26      @Override
27      protected RedisFuture<?> execute(BaseRedisAsyncCommands<String, String> commands, T item, String key, String value) {
28          Double score = this.score.convert(item);
29          if (score == null) {
30              return null;
31          }
32          if (payload == null && !increment) {
33              return ((RedisModulesAsyncCommands<String, String>) commands).sugadd(key, value, score);
34          }
35          SugaddOptions<String> options = new SugaddOptions<>();
36          options.setIncrement(increment);
37          if (payload != null) {
38              options.setPayload(payload.convert(item));
39          }
40          return ((RedisModulesAsyncCommands<String, String>) commands).sugadd(key, value, score, options);
41      }
42  
43  }
