1   package com.redislabs.riot.redis;
2   
3   import org.springframework.batch.item.redis.OperationItemWriter;
4   import picocli.CommandLine.Command;
5   import picocli.CommandLine.Option;
6   
7   import java.util.Map;
8   
9   @Command(name = "sugadd", description = "Add suggestion strings to a RediSearch auto-complete suggestion dictionary")
10  public class SugaddCommand extends AbstractKeyCommand {
11  
12      @Option(names = "--field", required = true, description = "Field containing the strings to add", paramLabel = "<field>")
13      private String field;
14      @SuppressWarnings("unused")
15      @Option(names = "--score", description = "Name of the field to use for scores", paramLabel = "<field>")
16      private String scoreField;
17      @Option(names = "--score-default", description = "Score when field not present (default: ${DEFAULT-VALUE})", paramLabel = "<num>")
18      private double scoreDefault = 1;
19      @Option(names = "--payload", description = "Field containing the payload", paramLabel = "<field>")
20      private String payload;
21      @Option(names = "--increment", description = "Increment the existing suggestion by the score instead of replacing the score")
22      private boolean increment;
23  
24      @Override
25      public OperationItemWriter.RedisOperation<Map<String, Object>> operation() {
26          return new Sugadd<>(key(), stringFieldExtractor(field), numberFieldExtractor(Double.class, scoreField, scoreDefault), stringFieldExtractor(payload), increment);
27      }
28  
29  
30  }
