3   import io.lettuce.core.XAddArgs;
4   import org.springframework.batch.item.redis.OperationItemWriter;

6   import picocli.CommandLine;

9   
10  import java.util.Map;
11  
12  @Command(name = "xadd", description = "Append entries to a stream")
13  public class XaddCommand extends AbstractKeyCommand {
14  
15      @CommandLine.Mixin
16      private FilteringOptionsilteringOptions">FilteringOptions filteringOptions = new FilteringOptions();
17      @SuppressWarnings("unused")
18      @Option(names = "--maxlen", description = "Stream maxlen", paramLabel = "<int>")
19      private Long maxlen;
20      @SuppressWarnings("unused")
21      @Option(names = "--trim", description = "Stream efficient trimming ('~' flag)")
22      private boolean approximateTrimming;
23  
24      @Override
25      public OperationItemWriter.RedisOperation<Map<String, Object>> operation() {
26          XAddArgs args = new XAddArgs();
27          if (maxlen != null) {
28              args.maxlen(maxlen);
29          }
30          args.approximateTrimming(approximateTrimming);
31          return new Xadd<>(key(), filteringOptions.converter(), args);
32      }
