3   import com.redislabs.riot.RedisOptions;
4   import com.redislabs.riot.RiotStepBuilder;
5   import lombok.Data;
6   import lombok.EqualsAndHashCode;
7   import lombok.extern.slf4j.Slf4j;
8   import org.springframework.batch.core.configuration.annotation.StepBuilderFactory;
9   import org.springframework.batch.core.job.flow.Flow;
10  import org.springframework.batch.core.job.flow.support.SimpleFlow;
11  import org.springframework.batch.core.step.builder.StepBuilder;
12  import org.springframework.batch.core.step.tasklet.TaskletStep;
13  import org.springframework.batch.item.ItemReader;
14  import org.springframework.batch.item.ItemWriter;
15  import org.springframework.batch.item.redis.support.FlushingStepBuilder;
16  import org.springframework.batch.item.redis.support.KeyValue;
17  import org.springframework.batch.item.redis.support.KeyValueItemReader;
18  import org.springframework.batch.item.redis.support.PollableItemReader;
19  import org.springframework.core.task.SimpleAsyncTaskExecutor;
20  import picocli.CommandLine;
21  
22  @Data
23  @EqualsAndHashCode(callSuper = true)
24  @Slf4j
25  public abstract class AbstractReplicateCommand<T extends KeyValue<?>> extends AbstractTargetCommand {
26  
27      @SuppressWarnings("unused")
28      @CommandLine.Mixin
29      private ReplicationOptionslicationOptions">ReplicationOptions replicationOptions = new ReplicationOptions();
30  
31      @Override
32      protected Flow flow(StepBuilderFactory stepBuilderFactory) {
33          if (replicationOptions.isVerify()) {
34              return flow("replication-verification-flow").start(replicationFlow(stepBuilderFactory)).next(verificationFlow(stepBuilderFactory)).build();
35          }
36          return replicationFlow(stepBuilderFactory);
37      }
38  
39      private Flow replicationFlow(StepBuilderFactory stepBuilderFactory) {
40          switch (replicationOptions.getMode()) {
41              case LIVE:
42                  SimpleFlow notificationFlow = flow("notification-flow").start(notificationStep(stepBuilderFactory).build()).build();
43                  SimpleFlow scanFlow = flow("scan-flow").start(scanStep(stepBuilderFactory)).build();
44                  return flow("live-flow").split(new SimpleAsyncTaskExecutor()).add(notificationFlow, scanFlow).build();
45              case LIVEONLY:
46                  return flow("live-only-flow").start(notificationStep(stepBuilderFactory).build()).build();
47              default:
48                  return flow("snapshot-flow").start(scanStep(stepBuilderFactory)).build();
49          }
50      }
51  
52      private TaskletStep scanStep(StepBuilderFactory stepBuilderFactory) {
53          StepBuilder stepBuilder = stepBuilderFactory.get("scan-replication-step");
54          RiotStepBuilder<T, T> scanStep = riotStep(stepBuilder, "Scanning");
55          initialMax(scanStep);
56          return scanStep.reader(reader(getRedisOptions())).writer(writer(targetRedisOptions)).build().build();
57      }
58  
59      private FlushingStepBuilder<T, T> notificationStep(StepBuilderFactory stepBuilderFactory) {
60          StepBuilder stepBuilder = stepBuilderFactory.get("live-replication-step");
61          RiotStepBuilder<T, T> notificationStep = riotStep(stepBuilder, "Listening");
62          return configure(notificationStep.reader(liveReader(getRedisOptions())).writer(writer(targetRedisOptions)).build());
63      }
64  
65      protected abstract ItemReader<T> reader(RedisOptions redisOptions);
66  
67      protected abstract PollableItemReader<T> liveReader(RedisOptions redisOptions);
68  
69      protected abstract ItemWriter<T> writer(RedisOptions redisOptions);
70  
71      @SuppressWarnings({"rawtypes", "unchecked"})
72      protected <B extends KeyValueItemReader.LiveKeyValueItemReaderBuilder> B configure(B builder) {
73          log.debug("Configuring live reader with {}, queueCapacity={}", readerOptions, replicationOptions.getNotificationQueueCapacity());
74          return (B) readerOptions.configure(builder.keyPattern(readerOptions.getScanMatch()).queueCapacity(replicationOptions.getNotificationQueueCapacity()).database(getRedisOptions().uris().get(0).getDatabase()).flushingInterval(flushingTransferOptions.getFlushIntervalDuration()).idleTimeout(flushingTransferOptions.getIdleTimeoutDuration()));
75      }

77  }
