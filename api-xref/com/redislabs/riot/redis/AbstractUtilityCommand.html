1   package com.redislabs.riot.redis;
2   
3   import com.redislabs.riot.AbstractTaskCommand;
4   import com.redislabs.riot.RedisOptions;
5   import io.lettuce.core.AbstractRedisClient;
6   import io.lettuce.core.RedisClient;
7   import io.lettuce.core.api.StatefulConnection;
8   import io.lettuce.core.api.StatefulRedisConnection;
9   import io.lettuce.core.api.sync.BaseRedisCommands;
10  import io.lettuce.core.cluster.RedisClusterClient;
11  import io.lettuce.core.cluster.api.StatefulRedisClusterConnection;
12  import org.springframework.batch.core.configuration.annotation.StepBuilderFactory;
13  import org.springframework.batch.core.job.flow.Flow;
14  import org.springframework.batch.repeat.RepeatStatus;
15  import org.springframework.util.ClassUtils;
16  import picocli.CommandLine.Command;
17  
18  @Command
19  public abstract class AbstractUtilityCommand extends AbstractTaskCommand {
20  
21      @Override
22      protected Flow flow(StepBuilderFactory stepBuilderFactory) {
23  
24          return flow(stepBuilderFactory.get(ClassUtils.getShortName(getClass()) + "-step").tasklet((contribution, chunkContext) -> {
25              RedisOptions redisOptions = getRedisOptions();
26              AbstractRedisClient client = redisOptions.client();
27              try {
28                  try (StatefulConnection<String, String> connection = redisOptions.isCluster() ? ((RedisClusterClient) client).connect() : ((RedisClient) client).connect()) {
29                      BaseRedisCommands<String, String> commands = redisOptions.isCluster() ? ((StatefulRedisClusterConnection<String, String>) connection).sync() : ((StatefulRedisConnection<String, String>) connection).sync();
30                      execute(commands);
31                      return RepeatStatus.FINISHED;
32                  }
33              } finally {
34                  RedisOptions.shutdown(client);
35              }
36          }).build());
37      }
38  
39      protected abstract void execute(BaseRedisCommands<String, String> commands);
40  
41  
42  }
