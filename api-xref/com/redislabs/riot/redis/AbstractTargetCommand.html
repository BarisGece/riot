41      protected Flow verificationFlow(StepBuilderFactory stepBuilderFactory) {
42          KeyValueItemReader<DataStructure> sourceReader = dataStructureReader();
43          log.debug("Creating key comparator with TTL tolerance of {} seconds", compareOptions.getTtlTolerance());
44          DataStructureValueReader targetValueReader = targetDataStructureValueReader();
45          KeyComparisonResultCounter counter = new KeyComparisonResultCounter();
46          KeyComparisonItemWriter.KeyComparisonItemWriterBuilder writerBuilder = KeyComparisonItemWriter.valueReader(targetValueReader);
47          writerBuilder.resultHandler(counter);
48          if (compareOptions.isShowDiffs()) {
49              writerBuilder.resultHandler(this::log);
50          }
51          writerBuilder.ttlTolerance(compareOptions.getTtlToleranceDuration());
52          KeyComparisonItemWriter writer = writerBuilder.build();
53          StepBuilder verificationStepBuilder = stepBuilderFactory.get("verification-step");
54          RiotStepBuilder<DataStructure, DataStructure> stepBuilder = riotStep(verificationStepBuilder, "Verifying");
55          initialMax(stepBuilder);
56          stepBuilder.reader(sourceReader).writer(writer);
57          stepBuilder.extraMessage(() -> extraMessage(counter));
58          SimpleStepBuilder<DataStructure, DataStructure> step = stepBuilder.build();
59          step.listener(new StepExecutionListenerSupport() {
60              @Override
61              public ExitStatus afterStep(StepExecution stepExecution) {
62                  if (counter.isOK()) {
63                      log.info("Verification completed - all OK");
64                      return super.afterStep(stepExecution);
65                  }
66                  try {
67                      Thread.sleep(transferOptions.getProgressUpdateIntervalMillis());
68                  } catch (InterruptedException e) {
69                      log.debug("Verification interrupted");
70                      return null;
71                  }
72                  log.warn("Verification failed: identical={}, missing={}, extraneous={}, values={}, ttls={}, types={}", (Object[]) counter.get(KeyComparisonItemWriter.Result.OK, KeyComparisonItemWriter.Result.SOURCE, KeyComparisonItemWriter.Result.TARGET, KeyComparisonItemWriter.Result.VALUE, KeyComparisonItemWriter.Result.TTL, KeyComparisonItemWriter.Result.TYPE));
73                  return new ExitStatus(ExitStatus.FAILED.getExitCode(), "Verification failed");
74              }
75          });
76          TaskletStep verificationStep = step.build();
77          return flow("verification-flow").start(verificationStep).build();
78      }

80      private String extraMessage(KeyComparisonResultCounter counter) {
81          Long[] counts = counter.get(KeyComparisonItemWriter.Result.SOURCE, KeyComparisonItemWriter.Result.TYPE, KeyComparisonItemWriter.Result.VALUE, KeyComparisonItemWriter.Result.TTL, KeyComparisonItemWriter.Result.TARGET);
82          return " " + String.format(extraMessageFormat(), (Object[]) counts);
83      }
84  
85      private String extraMessageFormat() {
86          switch (transferOptions.getProgress()) {
87              case COLOR:
88                  return COLORFUL_COMPARE_MESSAGE_FORMAT;
89              default:
90                  return ASCII_COMPARE_MESSAGE_FORMAT;
91          }
92      }
93  
94      private void log(DataStructure source, DataStructure target, KeyComparisonItemWriter.Result result) {
95          switch (result) {
96              case OK:
97                  return;
98              case SOURCE:
99                  log.warn("Key {} is missing from target", source.getKey());
100                 return;
101             case TARGET:
102                 log.warn("Key {} in target but not in source", target.getKey());
103                 return;
104             case TTL:
105                 log.warn("Key {} has different TTLs: source={} target={}", source.getKey(), source.getAbsoluteTTL(), target.getAbsoluteTTL());
106                 return;
107             case TYPE:
108                 log.warn("Key {} has different types: source={} target={}", source.getKey(), source.getType(), target.getType());
109                 return;
110             case VALUE:
111                 log.warn("Key {} has different values: source={} target={}", source.getKey(), source.getValue(), target.getValue());
112                 return;
113         }
114     }
