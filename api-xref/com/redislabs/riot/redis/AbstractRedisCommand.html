1   package com.redislabs.riot.redis;
2   
3   import com.redislabs.riot.HelpCommand;
4   import com.redislabs.riot.RedisCommand;
5   import com.redislabs.riot.convert.CompositeConverter;
6   import com.redislabs.riot.convert.FieldExtractor;
7   import com.redislabs.riot.convert.ObjectToNumberConverter;
8   import com.redislabs.riot.convert.ObjectToStringConverter;
9   import lombok.Data;
10  import lombok.EqualsAndHashCode;
11  import org.springframework.batch.item.redis.support.KeyMaker;
12  import org.springframework.core.convert.converter.Converter;
13  import org.springframework.util.ObjectUtils;
14  import picocli.CommandLine;
15  
16  import java.util.Map;
17  
18  @Data
19  @EqualsAndHashCode(callSuper = true)
20  @CommandLine.Command(sortOptions = false, abbreviateSynopsis = true)
21  public abstract class AbstractRedisCommand<O> extends HelpCommand implements RedisCommand<O> {
22  
23      @CommandLine.Option(names = {"-s", "--separator"}, description = "Key separator (default: ${DEFAULT-VALUE})", paramLabel = "<str>")
24      private String keySeparator = ":";
25      @SuppressWarnings("unused")
26      @CommandLine.Option(names = {"-r", "--remove"}, description = "Remove key or member fields the first time they are used")
27      private boolean removeFields;
28  
29      protected Converter<Map<String, Object>, Double> doubleFieldExtractor(String field) {
30          return numberFieldExtractor(Double.class, field, null);
31      }
32  
33      protected Converter<Map<String, Object>, Object> fieldExtractor(String field, Object defaultValue) {
34          return FieldExtractor.builder().field(field).remove(removeFields).defaultValue(defaultValue).build();
35      }
36  
37      @SuppressWarnings({"unchecked", "rawtypes"})
38      protected Converter<Map<String, Object>, String> stringFieldExtractor(String field) {
39          Converter<Map<String, Object>, String> extractor = (Converter) fieldExtractor(field, null);
40          if (extractor == null) {
41              return null;
42          }
43          return new CompositeConverter(extractor, new ObjectToStringConverter());
44      }
45  
46      @SuppressWarnings("unchecked")
47      protected <T extends Number> Converter<Map<String, Object>, T> numberFieldExtractor(Class<T> targetType, String field, T defaultValue) {
48          Converter<Map<String, Object>, Object> extractor = fieldExtractor(field, defaultValue);
49          return new CompositeConverter(extractor, new ObjectToNumberConverter<>(targetType));
50      }
51  
52      @SuppressWarnings({"rawtypes", "unchecked"})
53      protected KeyMaker<Map<String, Object>> idMaker(String prefix, String[] fields) {
54          KeyMaker.KeyMakerBuilder<Map<String, Object>> builder = KeyMaker.<Map<String, Object>>builder().separator(keySeparator).prefix(prefix);
55          if (!ObjectUtils.isEmpty(fields)) {
56              Converter[] converters = new Converter[fields.length];
57              for (int index = 0; index < fields.length; index++) {
58                  Converter<Map<String, Object>, Object> extractor = FieldExtractor.builder().remove(removeFields).field(fields[index]).build();
59                  CompositeConverter converter = new CompositeConverter(extractor, new ObjectToStringConverter());
60                  converters[index] = converter;
61              }
62              builder.converters(converters);
63          }
64          return builder.build();
65      }
66  
67  }
