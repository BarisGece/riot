1   package com.redislabs.riot.redis;
2   
3   import com.redislabs.riot.convert.MapToStringArrayConverter;
4   import io.lettuce.core.ScriptOutputType;
5   import org.springframework.batch.item.redis.support.operation.Eval;
6   import org.springframework.core.convert.converter.Converter;
7   import org.springframework.util.ObjectUtils;
8   import picocli.CommandLine.Command;
9   import picocli.CommandLine.Option;
10  
11  import java.util.Map;
12  
13  @Command(name = "eval", description = "Evaluate a Lua script with keys and arguments from input")
14  public class EvalCommand extends AbstractRedisCommand<Map<String, Object>> {
15  
16      private final static String[] EMPTY_STRING = new String[0];
17  
18      @SuppressWarnings("unused")
19      @Option(names = "--sha", description = "Digest", paramLabel = "<sha>")
20      private String sha;
21      @Option(arity = "1..*", names = "--keys", description = "Key fields", paramLabel = "<names>")
22      private String[] keys;
23      @Option(arity = "1..*", names = "--args", description = "Arg fields", paramLabel = "<names>")
24      private String[] args;
25      @Option(names = "--output", description = "Output: ${COMPLETION-CANDIDATES} (default: ${DEFAULT-VALUE})", paramLabel = "<type>")
26      private ScriptOutputType outputType = ScriptOutputType.STATUS;
27  
28      @Override
29      public Eval<Map<String, Object>> operation() {
30          return new Eval<>(sha, outputType, converter(keys), converter(args));
31      }
32  
33      @SuppressWarnings("unchecked")
34      private Converter<Map<String, Object>, String[]> converter(String[] fields) {
35          if (ObjectUtils.isEmpty(fields)) {
36              return m -> EMPTY_STRING;
37          }
38          Converter<Map<String, Object>, String>[] extractors = new Converter[fields.length];
39          for (int index = 0; index < fields.length; index++) {
40              extractors[index] = stringFieldExtractor(fields[index]);
41          }
42          return new MapToStringArrayConverter(extractors);
43      }
44  
45  }
