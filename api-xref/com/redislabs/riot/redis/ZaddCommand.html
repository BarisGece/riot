1   package com.redislabs.riot.redis;
2   
3   import org.springframework.batch.item.redis.support.operation.Zadd;
4   import picocli.CommandLine.Command;
5   import picocli.CommandLine.Option;
6   
7   import java.util.Map;
8   
9   @Command(name = "zadd", description = "Add members with scores to a sorted set")
10  public class ZaddCommand extends AbstractCollectionCommand {
11  
12      @SuppressWarnings("unused")
13      @Option(names = "--score", description = "Name of the field to use for scores", paramLabel = "<field>")
14      private String scoreField;
15      @Option(names = "--score-default", description = "Score when field not present (default: ${DEFAULT-VALUE})", paramLabel = "<num>")
16      private double scoreDefault = 1;
17  
18      @Override
19      public Zadd<Map<String, Object>> operation() {
20          return new Zadd<>(key(), member(), numberFieldExtractor(Double.class, scoreField, scoreDefault));
21      }
22  
23  }
