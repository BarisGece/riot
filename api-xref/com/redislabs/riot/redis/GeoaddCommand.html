1   package com.redislabs.riot.redis;
2   
3   import org.springframework.batch.item.redis.OperationItemWriter;
4   import org.springframework.batch.item.redis.support.operation.Geoadd;
5   import picocli.CommandLine.Command;
6   import picocli.CommandLine.Option;
7   
8   import java.util.Map;
9   
10  @Command(name = "geoadd", description = "Add members to a geo set")
11  public class GeoaddCommand extends AbstractCollectionCommand {
12  
13      @SuppressWarnings("unused")
14      @Option(names = "--lon", description = "Longitude field", paramLabel = "<field>")
15      private String longitudeField;
16      @SuppressWarnings("unused")
17      @Option(names = "--lat", description = "Latitude field", paramLabel = "<field>")
18      private String latitudeField;
19  
20      @Override
21      public OperationItemWriter.RedisOperation<Map<String, Object>> operation() {
22          return new Geoadd<>(key(), member(), doubleFieldExtractor(longitudeField), doubleFieldExtractor(latitudeField));
23      }
24  
25  }
