1   package com.redislabs.riot;
2   
3   import com.redislabs.testcontainers.RedisClusterContainer;
4   import com.redislabs.testcontainers.RedisServer;
5   import io.lettuce.core.RedisURI;
6   import org.apache.commons.io.IOUtils;
7   import org.codehaus.plexus.util.cli.CommandLineUtils;
8   import picocli.CommandLine;
9   
10  import java.io.InputStream;
11  import java.nio.charset.Charset;
12  import java.util.function.Consumer;
13  
14  @SuppressWarnings("unchecked")
15  public abstract class AbstractRiotTest {
16  
17      protected abstract RiotApp app();
18  
19      @SuppressWarnings("ConstantConditions")
20      protected String[] args(String filename) throws Exception {
21          try (InputStream inputStream = getClass().getResourceAsStream("/" + filename)) {
22              String command = IOUtils.toString(inputStream, Charset.defaultCharset());
23              if (command.startsWith("riot-")) {
24                  command = command.substring(command.indexOf(" ") + 1);
25              }
26              return CommandLineUtils.translateCommandline(command);
27          }
28      }
29  
30      protected int execute(String filename, RedisServer container, Consumer<CommandLine.ParseResult>... configurators) throws Exception {
31          RiotApp app = app();
32          RiotCommandLine commandLine = app.commandLine();
33          CommandLine.ParseResult parseResult = commandLine.parseArgs(args(filename));
34          configure(app, container);
35          for (Consumer<CommandLine.ParseResult> configurator : configurators) {
36              configurator.accept(parseResult);
37          }
38          return commandLine.getExecutionStrategy().execute(parseResult);
39      }
40  
41      protected void configure(RiotApp app, RedisServer container) {
42          app.setInfo(true);
43          app.getRedisOptions().setUris(new RedisURI[]{RedisURI.create(container.getRedisURI())});
44          app.getRedisOptions().setCluster(container instanceof RedisClusterContainer);
45      }
46  
47  }
