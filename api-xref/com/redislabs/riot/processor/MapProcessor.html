1   package com.redislabs.riot.processor;
2   
3   import java.util.Map;
4   
5   import org.springframework.batch.item.ItemProcessor;
6   import org.springframework.core.convert.converter.Converter;
7   
8   public class MapProcessor implements ItemProcessor<Map<String, Object>, Map<String, Object>> {
9   
10  	private final Map<String, Converter<String, Map<String, String>>> extractors;
11  
12  	public MapProcessor(Map<String, Converter<String, Map<String, String>>> extractors) {
13  		this.extractors = extractors;
14  	}
15  
16  	@Override
17  	public Map<String, Object> process(Map<String, Object> item) {
18  		for (String field : extractors.keySet()) {
19  			Object value = item.get(field);
20  			if (value == null) {
21  				continue;
22  			}
23  			item.putAll(extractors.get(field).convert(value.toString()));
24  		}
25  		return item;
26  	}
27  
28  }
