1   package com.redislabs.riot.convert;
2   
3   import java.util.Iterator;
4   import java.util.LinkedHashMap;
5   import java.util.Map;
6   import java.util.Map.Entry;
7   
8   import org.springframework.core.convert.converter.Converter;
9   import org.springframework.lang.Nullable;
10  import org.springframework.util.StringUtils;
11  
12  /**
13   * Flattens a nested map using . and [] notation for key names
14   *
15   */
16  public class MapFlattener<T> implements Converter<Map<String, Object>, Map<String, T>> {
17  
18  	private final Converter<Object, T> elementConverter;
19  
20  	public MapFlattener(Converter<Object, T> elementConverter) {
21  		this.elementConverter = elementConverter;
22  	}
23  
24  	@Override
25  	public Map<String, T> convert(Map<String, Object> source) {
26  		Map<String, T> resultMap = new LinkedHashMap<>();
27  		flatten("", source.entrySet().iterator(), resultMap);
28  		return resultMap;
29  	}
30  
31  	private void flatten(String prefix, Iterator<? extends Entry<String, Object>> map, Map<String, T> flatMap) {
32  		String actualPrefix = StringUtils.hasText(prefix) ? prefix.concat(".") : prefix;
33  		while (map.hasNext()) {
34  			Entry<String, Object> element = map.next();
35  			flattenElement(actualPrefix.concat(element.getKey()), element.getValue(), flatMap);
36  		}
37  	}
38  
39  	@SuppressWarnings("unchecked")
40  	private void flattenElement(String propertyPrefix, @Nullable Object source, Map<String, T> flatMap) {
41  		if (source == null) {
42  			return;
43  		}
44  		if (source instanceof Iterable) {
45  			int counter = 0;
46  			for (Object element : (Iterable<Object>) source) {
47  				flattenElement(propertyPrefix + "[" + counter + "]", element, flatMap);
48  				counter++;
49  			}
50  		} else if (source instanceof Map) {
51  			flatten(propertyPrefix, ((Map<String, Object>) source).entrySet().iterator(), flatMap);
52  		} else {
53  			((Map<String, Object>) flatMap).put(propertyPrefix, elementConverter.convert(source));
54  		}
55  	}
56  
57  }
