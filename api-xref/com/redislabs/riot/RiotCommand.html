1   package com.redislabs.riot;
2   
3   import io.lettuce.core.RedisURI;
4   import lombok.Data;
5   import lombok.EqualsAndHashCode;
6   import lombok.extern.slf4j.Slf4j;
7   import picocli.CommandLine.Command;
8   import picocli.CommandLine.ParentCommand;
9   
10  import java.util.concurrent.Callable;
11  
12  @Slf4j
13  @Data
14  @EqualsAndHashCode(callSuper = true)
15  @Command(abbreviateSynopsis = true, sortOptions = false)
16  public abstract class RiotCommand extends HelpCommand implements Callable<Integer> {
17  
18      @SuppressWarnings("unused")
19      @ParentCommand
20      private RiotApp app;
21  
22      protected RedisOptions getRedisOptions() {
23          return app.getRedisOptions();
24      }
25  
26      protected String name(RedisURI redisURI) {
27          if (redisURI.getSocket() != null) {
28              return redisURI.getSocket();
29          }
30          if (redisURI.getSentinelMasterId() != null) {
31              return redisURI.getSentinelMasterId();
32          }
33          return redisURI.getHost();
34      }
35  
36  }
