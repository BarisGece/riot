106     public List<RedisURI> uris() {
107         List<RedisURI> redisURIs = new ArrayList<>();
108         if (ObjectUtils.isEmpty(uris)) {
109             RedisURI uri = RedisURI.create(host, port);
110             uri.setSocket(socket);
111             uri.setSsl(tls);
112             redisURIs.add(uri);
113         } else {
114             redisURIs.addAll(Arrays.asList(this.uris));
115         }
116         for (RedisURI uri : redisURIs) {
117             uri.setVerifyPeer(verifyPeer);
118             if (username != null) {
119                 uri.setUsername(username);
120             }
121             if (password != null) {
122                 uri.setPassword(password);
123             }
124             if (database != uri.getDatabase()) {
125                 uri.setDatabase(database);
126             }
127             if (timeout != uri.getTimeout().getSeconds()) {
128                 uri.setTimeout(Duration.ofSeconds(timeout));
129             }
130             if (clientName != null) {
131                 uri.setClientName(clientName);
132             }
133         }
134         return redisURIs;
135     }

137     private ClientResources clientResources() {
138         DefaultClientResources.Builder builder = DefaultClientResources.builder();
139         if (showMetrics) {
140             builder.commandLatencyRecorder(CommandLatencyCollector.create(DefaultCommandLatencyCollectorOptions.builder().enable().build()));
141             builder.commandLatencyPublisherOptions(DefaultEventPublisherOptions.builder().eventEmitInterval(Duration.ofSeconds(1)).build());
142             ClientResources resources = builder.build();
143             resources.eventBus().get().filter(redisEvent -> redisEvent instanceof CommandLatencyEvent).cast(CommandLatencyEvent.class).subscribe(e -> System.out.println(e.getLatencies()));
144         }
145         return builder.build();
146     }
147 
148     private SslOptions sslOptions() {
149         SslOptions.Builder builder = SslOptions.builder();
150         if (keystore != null) {
151             if (keystorePassword == null) {
152                 builder.keystore(keystore);
153             } else {
154                 builder.keystore(keystore, keystorePassword.toCharArray());
155             }
156         }
157         if (truststore != null) {
158             if (truststorePassword == null) {
159                 builder.truststore(truststore);
160             } else {
161                 builder.truststore(truststore, truststorePassword);
162             }
163         }
164         if (cert != null) {
165             builder.trustManager(cert);
166         }
167         return builder.build();
168     }

170     public RedisClusterClient redisClusterClient() {
171         RedisClusterClient client = RedisClusterClient.create(clientResources(), uris());
172         client.setOptions(ClusterClientOptions.builder().autoReconnect(autoReconnect).sslOptions(sslOptions()).build());
173         return client;
174     }

176     public RedisModulesClient redisClient() {
177         RedisModulesClient client = RedisModulesClient.create(clientResources(), uris().get(0));
178         client.setOptions(ClientOptions.builder().autoReconnect(autoReconnect).sslOptions(sslOptions()).build());
179         return client;
180     }
