1   package com.redislabs.riot;
2   
3   import com.redislabs.riot.redis.AbstractRedisCommand;
4   import lombok.Getter;
5   import picocli.CommandLine;
6   
7   import java.util.List;
8   
9   public class RiotCommandLine extends CommandLine {
10  
11      private final RiotApp app;
12      private final IExecutionStrategy executionStrategy;
13  
14      public RiotCommandLine(RiotApp app, IExecutionStrategy executionStrategy) {
15          super(app);
16          this.app = app;
17          this.executionStrategy = executionStrategy;
18      }
19  
20      @SuppressWarnings({"unchecked", "rawtypes"})
21      @Override
22      public ParseResult parseArgs(String... args) {
23          ParseResult parseResult = super.parseArgs(args);
24          ParseResult subcommand = parseResult.subcommand();
25          if (subcommand != null) {
26              Object command = subcommand.commandSpec().userObject();
27              if (AbstractImportCommand.class.isAssignableFrom(command.getClass())) {
28                  AbstractImportCommand<?, ?> importCommand = (AbstractImportCommand<?, ?>) command;
29                  List<ParseResult> parsedRedisCommands = subcommand.subcommands();
30                  for (ParseResult parsedRedisCommand : parsedRedisCommands) {
31                      if (parsedRedisCommand.isUsageHelpRequested()) {
32                          return parsedRedisCommand;
33                      }
34                      importCommand.getRedisCommands().add((AbstractRedisCommand) parsedRedisCommand.commandSpec().userObject());
35                  }
36                  setExecutionStrategy(executionStrategy);
37                  return subcommand;
38              }
39          }
40          return parseResult;
41      }
42  }
