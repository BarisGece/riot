1   package com.redislabs.riot.file;
2   
3   import com.amazonaws.services.s3.AmazonS3;
4   import com.amazonaws.services.s3.AmazonS3Client;
5   import com.amazonaws.services.s3.AmazonS3ClientBuilder;
6   import lombok.Data;
7   import org.springframework.cloud.aws.core.io.s3.SimpleStorageProtocolResolver;
8   import org.springframework.core.io.DefaultResourceLoader;
9   import org.springframework.core.io.Resource;
10  import picocli.CommandLine.Option;
11  
12  @Data
13  public class S3Options {
14  
15      @Option(names = "--s3-access", description = "Access key", paramLabel = "<key>")
16      private String accessKey;
17      @Option(names = "--s3-secret", arity = "0..1", interactive = true, description = "Secret key", paramLabel = "<key>")
18      private String secretKey;
19      @Option(names = "--s3-region", description = "AWS region", paramLabel = "<name>")
20      private String region;
21  
22      public Resource resource(String location) {
23          AmazonS3ClientBuilder clientBuilder = AmazonS3Client.builder();
24          if (region != null) {
25              clientBuilder.withRegion(region);
26          }
27          if (accessKey != null) {
28              clientBuilder.withCredentials(new SimpleAWSCredentialsProvider(accessKey, secretKey));
29          }
30          SimpleStorageProtocolResolver resolver = new SimpleStorageProtocolResolver() {
31              @Override
32              public AmazonS3 getAmazonS3() {
33                  return clientBuilder.build();
34              }
35          };
36          resolver.afterPropertiesSet();
37          return resolver.resolve(location, new DefaultResourceLoader());
38      }
39  }
