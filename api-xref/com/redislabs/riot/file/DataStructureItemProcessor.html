1   package com.redislabs.riot.file;
2   
3   import io.lettuce.core.ScoredValue;
4   import io.lettuce.core.StreamMessage;
5   import org.springframework.batch.item.ItemProcessor;
6   import org.springframework.batch.item.redis.support.DataStructure;
7   
8   import java.util.ArrayList;
9   import java.util.Collection;
10  import java.util.Map;
11  
12  public class DataStructureItemProcessor implements ItemProcessor<DataStructure, DataStructure> {
13  
14      @SuppressWarnings("unchecked")
15      @Override
16      public DataStructure process(DataStructure item) throws Exception {
17          String type = item.getType().toLowerCase();
18          if (type.equals(DataStructure.ZSET)) {
19              Collection<Map<String, Object>> zset = (Collection<Map<String, Object>>) item.getValue();
20              Collection<ScoredValue<String>> values = new ArrayList<>(zset.size());
21              for (Map<String, Object> map : zset) {
22                  double score = ((Number) map.get("score")).doubleValue();
23                  String value = (String) map.get("value");
24                  values.add((ScoredValue<String>) ScoredValue.fromNullable(score, value));
25              }
26              item.setValue(values);
27          } else if (type.equals(DataStructure.STREAM)) {
28              Collection<Map<String, Object>> stream = (Collection<Map<String, Object>>) item.getValue();
29              Collection<StreamMessage<String, String>> messages = new ArrayList<>(stream.size());
30              for (Map<String, Object> message : stream) {
31                  messages.add(new StreamMessage<>((String) message.get("stream"), (String) message.get("id"), (Map<String, String>) message.get("body")));
32              }
33              item.setValue(messages);
34          }
35          return item;
36      }
37  
38  }
