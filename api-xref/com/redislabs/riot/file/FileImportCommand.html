55      @Override
56      protected Flow flow(StepBuilderFactory stepBuilderFactory) throws Exception {
57          Assert.isTrue(!ObjectUtils.isEmpty(files), "No file specified");
58          List<String> expandedFiles = FileUtils.expand(files);
59          if (ObjectUtils.isEmpty(expandedFiles)) {
60              throw new FileNotFoundException("File not found: " + String.join(", ", files));
61          }
62          List<Step> steps = new ArrayList<>();
63          for (String file : expandedFiles) {
64              FileType fileType = type(file);
65              if (fileType == null) {
66                  throw new IllegalArgumentException("Could not determine type of file " + file);
67              }
68              Resource resource = options.inputResource(file);
69              AbstractItemStreamItemReader<Map<String, Object>> reader = reader(file, fileType, resource);
70              reader.setName(file + "-reader");
71              StepBuilder stepBuilder = stepBuilderFactory.get(file + "-file-import-step");
72              steps.add(step(stepBuilder, "Importing " + file, reader).build());
73          }
74          return flow(steps.toArray(new Step[0]));
75      }

77      private FileType type(String file) {
78          if (type == null) {
79              String extension = FileUtils.extension(file);
80              if (extension != null) {
81                  switch (extension.toLowerCase()) {
82                      case FileUtils.EXTENSION_CSV:
83                      case FileUtils.EXTENSION_PSV:
84                      case FileUtils.EXTENSION_TSV:
85                          return FileType.DELIMITED;
86                      case FileUtils.EXTENSION_FW:
87                          return FileType.FIXED;
88                      case FileUtils.EXTENSION_JSON:
89                          return FileType.JSON;
90                      case FileUtils.EXTENSION_XML:
91                          return FileType.XML;
92                  }
93              }
94              return null;
95          }
96          return type;
97      }

99      @SuppressWarnings({"unchecked", "rawtypes"})
100     private AbstractItemStreamItemReader<Map<String, Object>> reader(String file, FileType fileType, Resource resource) {
101         switch (fileType) {
102             case DELIMITED:
103                 DelimitedLineTokenizer tokenizer = new DelimitedLineTokenizer();
104                 tokenizer.setDelimiter(delimiter(file));
105                 tokenizer.setQuoteCharacter(options.getQuoteCharacter());
106                 if (!ObjectUtils.isEmpty(options.getIncludedFields())) {
107                     tokenizer.setIncludedFields(options.getIncludedFields());
108                 }
109                 log.debug("Creating delimited reader with {} for file {}", options, file);
110                 return flatFileReader(resource, tokenizer);
111             case FIXED:
112                 FixedLengthTokenizer fixedLengthTokenizer = new FixedLengthTokenizer();
113                 RangeArrayPropertyEditor editor = new RangeArrayPropertyEditor();
114                 Assert.notEmpty(options.getColumnRanges(), "Column ranges are required");
115                 editor.setAsText(String.join(",", options.getColumnRanges()));
116                 Range[] ranges = (Range[]) editor.getValue();
117                 if (ranges.length == 0) {
118                     throw new IllegalArgumentException("Invalid ranges specified: " + Arrays.toString(options.getColumnRanges()));
119                 }
120                 fixedLengthTokenizer.setColumns(ranges);
121                 log.debug("Creating fixed-width reader with {} for file {}", options, file);
122                 return flatFileReader(resource, fixedLengthTokenizer);
123             case XML:
124                 log.debug("Creating XML reader for file {}", file);
125                 return (XmlItemReader) FileUtils.xmlReader(resource, Map.class);
126             default:
127                 log.debug("Creating JSON reader for file {}", file);
128                 return (JsonItemReader) FileUtils.jsonReader(resource, Map.class);
129         }
130     }

132     private String delimiter(String file) {
133         if (options.getDelimiter() == null) {
134             String extension = FileUtils.extension(file);
135             if (extension != null) {
136                 switch (extension.toLowerCase()) {
137                     case FileUtils.EXTENSION_CSV:
138                         return DelimitedLineTokenizer.DELIMITER_COMMA;
139                     case FileUtils.EXTENSION_PSV:
140                         return DELIMITER_PIPE;
141                     case FileUtils.EXTENSION_TSV:
142                         return DelimitedLineTokenizer.DELIMITER_TAB;
143                 }
144             }
145             throw new IllegalArgumentException("Could not determine delimiter for extension " + extension);
146         }
147         return options.getDelimiter();
148     }

155     private FlatFileItemReader<Map<String, Object>> flatFileReader(Resource resource, AbstractLineTokenizer tokenizer) {
156         if (!ObjectUtils.isEmpty(options.getNames())) {
157             tokenizer.setNames(options.getNames());
158         }
159         FlatFileItemReaderBuilder<Map<String, Object>> builder = new FlatFileItemReaderBuilder<>();
160         builder.resource(resource);
161         builder.encoding(options.getEncoding().name());
162         builder.lineTokenizer(tokenizer);
163         builder.recordSeparatorPolicy(recordSeparatorPolicy());
164         builder.linesToSkip(options.linesToSkip());
165         builder.strict(true);
166         builder.saveState(false);
167         builder.fieldSetMapper(new MapFieldSetMapper());
168         builder.skippedLinesCallback(new HeaderCallbackHandler(tokenizer));
169         return builder.build();
170     }
171 
172     private RecordSeparatorPolicy recordSeparatorPolicy() {
173         return new DefaultRecordSeparatorPolicy(options.getQuoteCharacter().toString(), options.getContinuationString());
174     }
