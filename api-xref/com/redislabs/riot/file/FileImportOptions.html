1   package com.redislabs.riot.file;
2   
3   import lombok.Data;
4   import lombok.EqualsAndHashCode;
5   import org.springframework.batch.item.file.transform.DelimitedLineTokenizer;
6   import picocli.CommandLine.Option;
7   
8   @Data
9   @EqualsAndHashCode(callSuper = true)
10  public class FileImportOptions extends FileOptions {
11  
12      public static final String DEFAULT_CONTINUATION_STRING = "\\";
13  
14      @Option(names = "--fields", arity = "1..*", description = "Delimited/FW field names", paramLabel = "<names>")
15      private String[] names;
16      @Option(names = {"-h", "--header"}, description = "Delimited/FW first line contains field names")
17      private boolean header;
18      @Option(names = "--delimiter", description = "Delimiter character", paramLabel = "<string>")
19      private String delimiter;
20      @Option(names = "--skip", description = "Delimited/FW lines to skip at start", paramLabel = "<count>")
21      private Integer linesToSkip;
22      @Option(names = "--include", arity = "1..*", description = "Delimited/FW field indices to include (0-based)", paramLabel = "<index>")
23      private int[] includedFields;
24      @Option(names = "--ranges", arity = "1..*", description = "Fixed-width column ranges", paramLabel = "<string>")
25      private String[] columnRanges;
26      @Option(names = "--quote", description = "Escape character for delimited files (default: ${DEFAULT-VALUE})", paramLabel = "<char>")
27      private Character quoteCharacter = DelimitedLineTokenizer.DEFAULT_QUOTE_CHARACTER;
28      @Option(names = "--cont", description = "Line continuation string (default: ${DEFAULT-VALUE})", paramLabel = "<string>")
29      private String continuationString = DEFAULT_CONTINUATION_STRING;
30  
31      public int linesToSkip() {
32          if (linesToSkip == null) {
33              if (header) {
34                  return 1;
35              }
36              return 0;
37          }
38          return linesToSkip;
39      }
40  
41  }
