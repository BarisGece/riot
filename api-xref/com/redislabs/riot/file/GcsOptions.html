1   package com.redislabs.riot.file;
2   
3   import com.google.auth.oauth2.GoogleCredentials;
4   import com.google.cloud.ServiceOptions;
5   import com.google.cloud.storage.StorageOptions;
6   import lombok.Data;
7   import org.springframework.cloud.gcp.autoconfigure.storage.GcpStorageAutoConfiguration;
8   import org.springframework.cloud.gcp.core.GcpScope;
9   import org.springframework.cloud.gcp.core.UserAgentHeaderProvider;
10  import org.springframework.cloud.gcp.storage.GoogleStorageResource;
11  import picocli.CommandLine.Option;
12  
13  import java.io.ByteArrayInputStream;
14  import java.io.File;
15  import java.io.IOException;
16  import java.nio.file.Files;
17  import java.util.Base64;
18  
19  @Data
20  public class GcsOptions {
21  
22      @Option(names = "--gcs-key-file", description = "GCS private key (e.g. /usr/local/key.json)", paramLabel = "<file>")
23      private File credentials;
24      @Option(names = "--gcs-project", description = "GCP project id", paramLabel = "<id>")
25      private String projectId;
26      @Option(names = "--gcs-key", arity = "0..1", interactive = true, description = "GCS Base64 encoded key", paramLabel = "<key>")
27      private String encodedKey;
28  
29      public GoogleStorageResource resource(String locationUri, boolean readOnly) throws IOException {
30          StorageOptions.Builder builder = StorageOptions.newBuilder().setProjectId(ServiceOptions.getDefaultProjectId()).setHeaderProvider(new UserAgentHeaderProvider(GcpStorageAutoConfiguration.class));
31          if (credentials != null) {
32              builder.setCredentials(GoogleCredentials.fromStream(Files.newInputStream(credentials.toPath())).createScoped((readOnly ? GcpScope.STORAGE_READ_ONLY : GcpScope.STORAGE_READ_WRITE).getUrl()));
33          }
34          if (encodedKey != null) {
35              builder.setCredentials(GoogleCredentials.fromStream(new ByteArrayInputStream(Base64.getDecoder().decode(encodedKey))));
36          }
37          if (projectId != null) {
38              builder.setProjectId(projectId);
39          }
40          return new GoogleStorageResource(builder.build().getService(), locationUri);
41      }
42  
43  }
