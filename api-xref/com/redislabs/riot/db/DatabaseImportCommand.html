21  @Slf4j
22  @Data
23  @EqualsAndHashCode(callSuper = true)
24  @Command(name = "import", description = "Import from a database")
25  public class DatabaseImportCommand extends AbstractImportCommand<Map<String, Object>, Map<String, Object>> {
26  
27      @CommandLine.Parameters(arity = "1", description = "SQL SELECT statement", paramLabel = "SQL")
28      private String sql;
29      @Mixin
30      private DataSourceOptionsourceOptions">DataSourceOptions dataSourceOptions = new DataSourceOptions();
31      @Mixin
32      private DatabaseImportOptionsatabaseImportOptions">DatabaseImportOptions importOptions = new DatabaseImportOptions();
33      @CommandLine.ArgGroup(exclusive = false, heading = "Processor options%n")
34      private ProcessorOptionsorOptions">ProcessorOptions processorOptions = new ProcessorOptions();
35  
36      @Override
37      protected Flow flow(StepBuilderFactory stepBuilderFactory) throws Exception {
38          log.debug("Creating data source: {}", dataSourceOptions);
39          DataSource dataSource = dataSourceOptions.dataSource();
40          String name = dataSource.getConnection().getMetaData().getDatabaseProductName();
41          log.debug("Creating {} database reader: {}", name, importOptions);
42          JdbcCursorItemReaderBuilder<Map<String, Object>> builder = new JdbcCursorItemReaderBuilder<>();
43          builder.saveState(false);
44          builder.dataSource(dataSource);
45          if (importOptions.getFetchSize() != null) {
46              builder.fetchSize(importOptions.getFetchSize());
47          }
48          if (importOptions.getMaxRows() != null) {
49              builder.maxRows(importOptions.getMaxRows());
50          }
51          builder.name(name + "-database-reader");
52          if (importOptions.getQueryTimeout() != null) {
53              builder.queryTimeout(importOptions.getQueryTimeout());
54          }
55          builder.rowMapper(new ColumnMapRowMapper());
56          builder.sql(sql);
57          builder.useSharedExtendedConnection(importOptions.isUseSharedExtendedConnection());
58          builder.verifyCursorPosition(importOptions.isVerifyCursorPosition());
59          JdbcCursorItemReader<Map<String, Object>> reader = builder.build();
60          reader.afterPropertiesSet();
61          return flow(step(stepBuilderFactory.get(name + "-db-import-step"), "Importing from " + name, reader).build());
62      }
63  
64      @Override
65      protected ItemProcessor<Map<String, Object>, Map<String, Object>> processor() throws NoSuchMethodException {
66          return processorOptions.processor(getRedisOptions());
67      }
