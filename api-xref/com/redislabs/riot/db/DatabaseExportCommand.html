26      @CommandLine.Parameters(arity = "1", description = "SQL INSERT statement.", paramLabel = "SQL")

30      @Mixin
31      private DatabaseExportOptionsatabaseExportOptions">DatabaseExportOptions exportOptions = new DatabaseExportOptions();

33      @Override
34      protected Flow flow(StepBuilderFactory stepBuilderFactory) throws Exception {
35          log.debug("Creating data source: {}", dataSourceOptions);
36          DataSource dataSource = dataSourceOptions.dataSource();
37          String name = dataSource.getConnection().getMetaData().getDatabaseProductName();
38          log.debug("Creating {} database writer: {}", name, exportOptions);
39          JdbcBatchItemWriterBuilder<Map<String, Object>> builder = new JdbcBatchItemWriterBuilder<>();
40          builder.itemSqlParameterSourceProvider(NullableMapSqlParameterSource::new);
41          builder.dataSource(dataSource);
42          builder.sql(sql);
43          builder.assertUpdates(exportOptions.isAssertUpdates());
44          JdbcBatchItemWriter<Map<String, Object>> writer = builder.build();
45          writer.afterPropertiesSet();
46          DataStructureItemProcessor processor = DataStructureItemProcessor.builder().keyRegex(exportOptions.getKeyRegex()).build();
47          return flow(step(stepBuilderFactory, processor, writer).build());
48      }
