18  import org.apache.commons.pool2.impl.GenericObjectPoolConfig;

29  @SuppressWarnings("unchecked")
30  @Testcontainers
31  public abstract class AbstractRiotIntegrationTest extends AbstractRiotTest {
32  
33      @Container
34      private static final RedisContainer REDIS = new RedisContainer().withKeyspaceNotifications();
35      @Container
36      private static final RedisClusterContainer REDIS_CLUSTER = new RedisClusterContainer().withKeyspaceNotifications();
37  
38      protected static final Map<RedisServer, AbstractRedisClient> CLIENTS = new HashMap<>();
39      protected static final Map<RedisServer, GenericObjectPool<? extends StatefulConnection<String, String>>> POOLS = new HashMap<>();
40      protected static final Map<RedisServer, StatefulConnection<String, String>> CONNECTIONS = new HashMap<>();
41      protected static final Map<RedisServer, StatefulRedisPubSubConnection<String, String>> PUBSUB_CONNECTIONS = new HashMap<>();
42      protected static final Map<RedisServer, BaseRedisAsyncCommands<String, String>> ASYNCS = new HashMap<>();
43      protected static final Map<RedisServer, BaseRedisCommands<String, String>> SYNCS = new HashMap<>();
44  
45      @BeforeAll
46      public static void setup() {
47          add(REDIS, REDIS_CLUSTER);
48      }
49  
50      private static void add(RedisServer... containers) {
51          for (RedisServer container : containers) {
52              if (container instanceof RedisClusterContainer) {
53                  RedisClusterClient client = RedisClusterClient.create(container.getRedisURI());
54                  CLIENTS.put(container, client);
55                  StatefulRedisClusterConnection<String, String> connection = client.connect();
56                  CONNECTIONS.put(container, connection);
57                  SYNCS.put(container, connection.sync());
58                  ASYNCS.put(container, connection.async());
59                  PUBSUB_CONNECTIONS.put(container, client.connectPubSub());
60                  POOLS.put(container, ConnectionPoolSupport.createGenericObjectPool(client::connect, new GenericObjectPoolConfig<>()));
61              } else {
62                  RedisClient client = RedisClient.create(container.getRedisURI());
63                  CLIENTS.put(container, client);
64                  StatefulRedisConnection<String, String> connection = client.connect();
65                  CONNECTIONS.put(container, connection);
66                  SYNCS.put(container, connection.sync());
67                  ASYNCS.put(container, connection.async());
68                  PUBSUB_CONNECTIONS.put(container, client.connectPubSub());
69                  POOLS.put(container, ConnectionPoolSupport.createGenericObjectPool(client::connect, new GenericObjectPoolConfig<>()));
70              }
71          }
72      }
73  
74      @AfterEach
75      public void flushall() {
76          for (BaseRedisCommands<String, String> sync : SYNCS.values()) {
77              ((RedisServerCommands<String, String>) sync).flushall();
78          }
79      }
80  
81      @AfterAll
82      public static void teardown() {
83          CONNECTIONS.values().forEach(RedisOptions::close);
84          PUBSUB_CONNECTIONS.values().forEach(RedisOptions::close);
85          POOLS.values().forEach(RedisOptions::close);
86          CLIENTS.values().forEach(RedisOptions::shutdown);
87          SYNCS.clear();
88          ASYNCS.clear();
89          CONNECTIONS.clear();
90          PUBSUB_CONNECTIONS.clear();
91          POOLS.clear();
92          CLIENTS.clear();
93      }
94  
95      static Stream<RedisServer> containers() {
96          return Stream.of(REDIS, REDIS_CLUSTER);
97      }
98  
99      static Stream<RedisContainer> standaloneContainer() {
100         return Stream.of(REDIS);
101     }
102 
103     protected static <T> T sync(RedisServer container) {
104         return (T) SYNCS.get(container);
105     }
106 
107     protected static <T> T async(RedisServer container) {
108         return (T) ASYNCS.get(container);
109     }
110 
111     protected static <C extends StatefulConnection<String, String>> C connection(RedisServer container) {
112         return (C) CONNECTIONS.get(container);
113     }
114 
115     protected static <C extends StatefulConnection<String, String>> GenericObjectPool<C> pool(RedisServer container) {
116         return (GenericObjectPool<C>) POOLS.get(container);
117     }
118 
119     protected DataGenerator.DataGeneratorBuilder dataGenerator(RedisServer container) {
120         return DataGenerator.connection(connection(container));
121     }
