54      private void configureLogging() {
55          InternalLoggerFactory.setDefaultFactory(JdkLoggerFactory.INSTANCE);
56          LogManager.getLogManager().reset();
57          Logger activeLogger = Logger.getLogger(ROOT_LOGGER);
58          ConsoleHandler handler = new ConsoleHandler();
59          handler.setLevel(Level.ALL);
60          handler.setFormatter(debug || stacktrace ? new StackTraceOneLineLogFormat() : new OneLineLogFormat());
61          activeLogger.addHandler(handler);
62          Logger.getLogger(ROOT_LOGGER).setLevel(logLevel());
63          Logger.getLogger("com.redislabs.riot").setLevel(riotLogLevel());
64          Logger.getLogger("org.springframework.batch.item.redis").setLevel(riotLogLevel());
65      }

80      private Level logLevel() {
81          if (debug) {
82              return Level.FINE;
83          }
84          if (info) {
85              return Level.INFO;
86          }
87          if (warning) {
88              return Level.SEVERE;
89          }
90          if (quiet) {
91              return Level.OFF;
92          }
93          return Level.WARNING;
94      }
95  
96      private Level riotLogLevel() {
97          if (debug) {
98              return Level.FINER;
99          }
100         if (info) {
101             return Level.FINE;
102         }
103         if (warning) {
104             return Level.WARNING;
105         }
106         if (quiet) {
107             return Level.SEVERE;
108         }
109         return Level.INFO;
110     }

112     private int handleExecutionException(Exception ex, CommandLine cmd, ParseResult parseResult) {
113         // bold red error message
114         cmd.getErr().println(cmd.getColorScheme().errorText(ex.getMessage()));
115         return cmd.getExitCodeExceptionMapper() != null ? cmd.getExitCodeExceptionMapper().getExitCode(ex) : cmd.getCommandSpec().exitCodeOnExecutionException();
116     }
