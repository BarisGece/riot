1   package com.redislabs.riot.gen;
2   
3   import lombok.Builder;
4   import org.springframework.batch.item.ItemReader;
5   import org.springframework.expression.EvaluationContext;
6   import org.springframework.expression.Expression;
7   import org.springframework.util.Assert;
8   
9   import java.util.Map;
10  
11  /**
12   * {@link ItemReader} that generates HashMaps using Faker.
13   *
14   * @author Julien Ruaux
15   */
16  public class FakerItemReader implements ItemReader<Map<String, Object>> {
17  
18      private final long start;
19      private final long end;
20      private final Generator<Map<String, Object>> generator;
21  
22      private EvaluationContext context;
23      private Map<String, Expression> expressions;
24      private long currentItemCount;
25  
26      @Builder
27      private FakerItemReader(Generator<Map<String, Object>> generator, long start, long end) {
28          Assert.notNull(generator, "A generator is required");
29          Assert.isTrue(end > start, "End index must be strictly greater than start index");
30          this.generator = generator;
31          this.start = start;
32          this.end = end;
33      }
34  
35      @Override
36      public Map<String, Object> read() {
37          if (currentItemCount >= end - start) {
38              return null;
39          }
40          currentItemCount++;
41          return generator.next(start + (currentItemCount % (end - start)));
42      }
43  
44  }
