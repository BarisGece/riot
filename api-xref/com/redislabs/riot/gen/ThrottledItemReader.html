1   package com.redislabs.riot.gen;
2   
3   import org.springframework.batch.item.ItemReader;
4   
5   public class ThrottledItemReader<T> implements ItemReader<T> {
6   
7       private final ItemReader<T> delegate;
8       private final long sleep;
9   
10      public ThrottledItemReader(ItemReader<T> delegate, long sleepDurationInMillis) {
11          this.delegate = delegate;
12          this.sleep = sleepDurationInMillis;
13      }
14  
15      @Override
16      public T read() throws Exception {
17          Thread.sleep(sleep);
18          return delegate.read();
19      }
20  }
