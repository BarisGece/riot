28  public class GeneratorImportCommand extends AbstractImportCommand<Map<String, Object>, Map<String, Object>> {

30      @CommandLine.Mixin
31      private GenerateOptions.html#GenerateOptions">GenerateOptions options = new GenerateOptions();
32      @CommandLine.ArgGroup(exclusive = false, heading = "Processor options%n")
33      private ProcessorOptionsorOptions">ProcessorOptions processorOptions = new ProcessorOptions();

35      @Override
36      protected Flow flow(StepBuilderFactory stepBuilderFactory) throws Exception {
37          StepBuilder stepBuilder = stepBuilderFactory.get("generate-step");
38          return flow(step(stepBuilder, "Generating", reader()).build());
39      }
40  
41      private ItemReader<Map<String, Object>> reader() {
42          log.debug("Creating Faker reader with {}", options);
43          FakerItemReader reader = FakerItemReader.builder().generator(generator()).start(options.getStart()).end(options.getEnd()).build();
44          if (options.getSleep() > 0) {
45              return new ThrottledItemReader<>(reader, options.getSleep());
46          }
47          return reader;
48      }
49  
50      private Generator<Map<String, Object>> generator() {
51          Map<String, String> fields = options.getFakerFields() == null ? new LinkedHashMap<>() : new LinkedHashMap<>(options.getFakerFields());
52          if (options.getFakerIndex() != null) {
53              fields.putAll(fieldsFromIndex(options.getFakerIndex()));
54          }
55          MapGenerator generator = MapGenerator.builder().locale(options.getLocale()).fields(fields).build();
56          if (options.isIncludeMetadata()) {
57              return new MapWithMetadataGenerator(generator);
58          }
59          return generator;
60      }
61  
62      private String expression(Field field) {
63          switch (field.getType()) {
64              case TEXT:
65                  return "lorem.paragraph";
66              case TAG:
67                  return "number.digits(10)";
68              case GEO:
69                  return "address.longitude.concat(',').concat(address.latitude)";
70              default:
71                  return "number.randomDouble(3,-1000,1000)";
72          }
73      }
74  
75      private Map<String, String> fieldsFromIndex(String index) {
76          Map<String, String> fields = new LinkedHashMap<>();
77          RedisModulesClient client = RedisModulesClient.create(getRedisOptions().uris().get(0));
78          try (StatefulRedisModulesConnection<String, String> connection = client.connect()) {
79              RediSearchCommands<String, String> commands = connection.sync();
80              IndexInfo info = RediSearchUtils.indexInfo(commands.indexInfo(index));
81              for (Field field : info.getFields()) {
82                  fields.put(field.getName(), expression(field));
83              }
84          } finally {
85              RedisOptions.shutdown(client);
86          }
87          return fields;
88      }
89

91      protected ItemProcessor<Map<String, Object>, Map<String, Object>> processor() throws NoSuchMethodException {
92          return processorOptions.processor(getRedisOptions());
93      }

99          return riotStepBuilder;
