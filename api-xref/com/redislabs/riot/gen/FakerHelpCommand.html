1   package com.redislabs.riot.gen;
2   
3   import com.github.javafaker.Faker;
4   import org.springframework.util.ClassUtils;
5   import picocli.CommandLine;
6   import picocli.CommandLine.Command;
7   import picocli.CommandLine.Help;
8   import picocli.CommandLine.IHelpCommandInitializable2;
9   
10  import java.io.PrintWriter;
11  import java.lang.reflect.Method;
12  import java.lang.reflect.Parameter;
13  import java.util.*;
14  import java.util.stream.Stream;
15  
16  @Command(name = "faker-help", header = "Displays help information about Faker", synopsisHeading = "%nUsage: ", helpCommand = true)
17  public class FakerHelpCommand implements IHelpCommandInitializable2, Runnable {
18  
19      @CommandLine.Parameters(description = "Name of the Faker provider to show help for", paramLabel = "<name>")
20      private String provider;
21  
22      private final static List<String> EXCLUDES = Arrays.asList("instance", "options");
23  
24      private PrintWriter outWriter;
25  
26      @Override
27      public void run() {
28          Stream<Method> methods = Arrays.stream(Faker.class.getDeclaredMethods()).filter(m -> !EXCLUDES.contains(m.getName())).filter(m -> m.getReturnType().getPackage().equals(Faker.class.getPackage()));
29          if (provider == null) {
30              outWriter.println("Run 'riot-gen faker-help <provider>' for documentation on a specific Faker provider:");
31              methods.sorted(Comparator.comparing(Method::getName)).forEach(p -> outWriter.println("  " + p.getName() + " " + url(ClassUtils.getShortName(p.getReturnType()))));
32          } else {
33              Optional<Method> method = methods.filter(m -> m.getName().equals(provider)).findFirst();
34              if (method.isPresent()) {
35                  Class<?> type = method.get().getReturnType();
36                  String shortName = ClassUtils.getShortName(type);
37                  outWriter.println("Available fakes: " + url(shortName));
38                  for (Method subMethod : type.getDeclaredMethods()) {
39                      outWriter.println("  " + provider + "." + methodToString(subMethod));
40                  }
41              } else {
42                  outWriter.println("No such field: " + provider);
43              }
44          }
45      }
46  
47      private String url(String name) {
48          return "http://dius.github.io/java-faker/apidocs/com/github/javafaker/" + name + ".html";
49      }
50  
51      private String methodToString(Method method) {
52          String result = method.getName();
53          Parameter[] parameters = method.getParameters();
54          if (parameters.length == 0) {
55              return result;
56          }
57          result += "(";
58          List<String> paramNames = new ArrayList<>();
59          for (Parameter parameter : parameters) {
60              paramNames.add(ClassUtils.getShortName(parameter.getType()));
61          }
62          result += String.join(",", paramNames);
63          result += ")";
64          return result;
65      }
66  
67      public void init(CommandLine helpCommandLine, Help.ColorScheme colorScheme, PrintWriter out, PrintWriter err) {
68          this.outWriter = notNull(out, "outWriter");
69      }
70  
71      static <T> T notNull(T object, String description) {
72          if (object == null) {
73              throw new NullPointerException(description);
74          }
75          return object;
76      }
77  
78  }
