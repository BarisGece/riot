1   package com.redislabs.riot.gen;
2   
3   import com.github.javafaker.Faker;
4   import lombok.Builder;
5   import lombok.NonNull;
6   import org.springframework.expression.Expression;
7   import org.springframework.expression.spel.standard.SpelExpressionParser;
8   import org.springframework.expression.spel.support.SimpleEvaluationContext;
9   import org.springframework.util.Assert;
10  
11  import java.util.HashMap;
12  import java.util.Locale;
13  import java.util.Map;
14  import java.util.stream.Collectors;
15  
16  public class MapGenerator implements Generator<Map<String,Object>> {
17  
18      public final static String FIELD_INDEX = "index";
19  
20      private final SimpleEvaluationContext context;
21      private final Map<String, Expression> expressions;
22  
23      @Builder
24      private MapGenerator(Locale locale, @NonNull Map<String, String> fields) {
25          Faker faker = new Faker(locale == null ? Locale.getDefault() : locale);
26          this.context = new SimpleEvaluationContext.Builder(new ReflectivePropertyAccessor()).withInstanceMethods().withRootObject(faker).build();
27          SpelExpressionParser parser = new SpelExpressionParser();
28          this.expressions = fields.entrySet().stream().collect(Collectors.toMap(Map.Entry::getKey, e -> parser.parseExpression(e.getValue())));
29      }
30  
31      @Override
32      public Map<String, Object> next(long index) {
33          Map<String, Object> map = new HashMap<>();
34          context.setVariable(FIELD_INDEX, index);
35          for (String field : expressions.keySet()) {
36              map.put(field, expressions.get(field).getValue(context));
37          }
38          return map;
39      }
40  }
