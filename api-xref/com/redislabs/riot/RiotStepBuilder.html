1   package com.redislabs.riot;
2   
3   import io.lettuce.core.RedisCommandExecutionException;
4   import lombok.Setter;
5   import lombok.experimental.Accessors;
6   import lombok.extern.slf4j.Slf4j;
7   import org.springframework.batch.core.ItemWriteListener;
8   import org.springframework.batch.core.StepExecutionListener;
9   import org.springframework.batch.core.step.builder.FaultTolerantStepBuilder;
10  import org.springframework.batch.core.step.builder.SimpleStepBuilder;
11  import org.springframework.batch.core.step.builder.StepBuilder;
12  import org.springframework.batch.item.ItemProcessor;
13  import org.springframework.batch.item.ItemReader;
14  import org.springframework.batch.item.ItemWriter;
15  import org.springframework.core.task.SyncTaskExecutor;
16  import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;
17  
18  import java.time.Duration;
19  import java.util.concurrent.ThreadPoolExecutor;
20  import java.util.function.Supplier;
21  
22  @Slf4j
23  @Setter
24  @Accessors(fluent = true)
25  public class RiotStepBuilder<I, O> {
26  
27      private final StepBuilder stepBuilder;
28      private final TransferOptions options;
29  
30      private String taskName;
31      private ItemReader<I> reader;
32      private ItemProcessor<I, O> processor;
33      private ItemWriter<O> writer;
34      private Supplier<String> extraMessage;
35      private Supplier<Long> initialMax;
36  
37      public RiotStepBuilder(StepBuilder stepBuilder, TransferOptions options) {
38          this.stepBuilder = stepBuilder;
39          this.options = options;
40      }
41  
42      @SuppressWarnings({"unchecked", "rawtypes"})
43      public SimpleStepBuilder<I, O> build() {
44          SimpleStepBuilder<I, O> step = stepBuilder.<I, O>chunk(options.getChunkSize()).reader(reader).processor(processor).writer(writer);
45          if (options.getProgress() != TransferOptions.Progress.NONE) {
46              ProgressMonitor.ProgressMonitorBuilder monitorBuilder = ProgressMonitor.builder();
47              monitorBuilder.style(options.getProgress());
48              monitorBuilder.taskName(taskName);
49              monitorBuilder.initialMax(initialMax);
50              monitorBuilder.updateInterval(Duration.ofMillis(options.getProgressUpdateIntervalMillis()));
51              monitorBuilder.extraMessage(extraMessage);
52              ProgressMonitor monitor = monitorBuilder.build();
53              step.listener((StepExecutionListener) monitor);
54              step.listener((ItemWriteListener) monitor);
55          }
56          FaultTolerantStepBuilder<I, O> ftStep = step.faultTolerant().skipLimit(options.getSkipLimit()).skip(RedisCommandExecutionException.class);
57          if (options.getThreads() > 1) {
58              ThreadPoolTaskExecutor taskExecutor = new ThreadPoolTaskExecutor();
59              taskExecutor.setCorePoolSize(options.getThreads());
60              taskExecutor.setMaxPoolSize(options.getThreads());
61              taskExecutor.setQueueCapacity(options.getThreads());
62              taskExecutor.setRejectedExecutionHandler(new ThreadPoolExecutor.CallerRunsPolicy());
63              taskExecutor.afterPropertiesSet();
64              log.debug("Created pooled task executor of size {}", taskExecutor.getCorePoolSize());
65              ftStep.taskExecutor(taskExecutor).throttleLimit(options.getThreads());
66          } else {
67              ftStep.taskExecutor(new SyncTaskExecutor());
68          }
69          return ftStep;
70      }
71  
72  }
